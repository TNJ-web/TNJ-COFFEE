  <script>
    const buttons = document.querySelectorAll('.shop-item button');
    const modal = document.getElementById('orderModal');
    const productInput = document.getElementById('productInput');
    const orderForm = document.getElementById('orderForm');
    const successMsg = document.getElementById('orderSuccess');
    const errorMsg = document.getElementById('orderError');

    buttons.forEach(button => {
      button.addEventListener('click', (e) => {
        const product = e.target.closest('.shop-item').querySelector('strong').innerText;
        productInput.value = product;
        modal.style.display = 'flex';
        orderForm.style.display = 'block';
        successMsg.style.display = 'none';
        errorMsg.style.display = 'none';
      });
    });

    function closeModal() {
      modal.style.display = 'none';
    }

    orderForm.addEventListener('submit', function(e) {
      e.preventDefault();
      const formData = new FormData(orderForm);

      fetch(orderForm.action, {
        method: 'POST',
        body: formData,
        headers: { 'Accept': 'application/json' }
      }).then(response => {
        if (response.ok) {
          orderForm.style.display = 'none';
          successMsg.style.display = 'block';
          errorMsg.style.display = 'none';
          setTimeout(() => {
            modal.style.display = 'none';
            successMsg.style.display = 'none';
            orderForm.reset();
          }, 5000);
        } else {
          successMsg.style.display = 'none';
          errorMsg.style.display = 'block';
        }
      }).catch(() => {
        successMsg.style.display = 'none';
        errorMsg.style.display = 'block';
      });
    });
  </script>
